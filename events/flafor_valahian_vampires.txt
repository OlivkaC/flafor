		dynasty = "Draculesti"

	}

			claim = 100
			no_consort_with_heir = yes
		}
		custom_tooltip = vampires_religion_takeover
		hidden_effect = {
			every_owned_province = {
				limit = { religion_group = christian }
				change_religion = vampires
			}
			capital_scope = {
				add_province_triggered_modifier = seat_of_dracula_own
				add_province_triggered_modifier = seat_of_dracula_foreign
				add_building = fort_16th
			}
		}
		add_country_modifier = { 
			name = return_of_vampires
			duration = -1 
		}
		
		if = {
			limit = { NOT = { government = monarchy } }
			
			change_government = monarchy
		}
		
		add_government_reform = vampires_ascedant
		swap_non_generic_missions = yes
		
		ai_chance = { 
			factor = 1
		}
	}
	option = {
		name = flafor_valahian_vampires.0.b
		exile_ruler_as = vam_exiled
		add_country_modifier = { 
			name = fear_of_vampires
			duration = 73000 
		}
	}
}

country_event = {
	id = flafor_valahian_vampires.2
	title = flafor_valahian_vampires.2.t
	desc = flafor_valahian_vampires.2.d
	picture = FAMINE_eventPicture
	
	mean_time_to_happen = {
		months = 24
	}
	
	trigger = {
		religion = vampires
		OR = {
			had_ruler_flag = {
				flag = improve_vampire
				days = 7300
			}
			NOT = { has_ruler_flag = improve_vampire }
		}
		OR = {
			NOT = { adm = 8 }
			NOT = { dip =  8 }
			NOT = { mil = 8 }
		}
	}
	
	immediate = {
		set_ruler_flag = improve_vampire
	}
	
	option = {
		name = flafor_valahian_vampires.2.a
		if = {
			limit = { adm = 8 }
			if = {
				limit = { dip = 8}
				change_mil = 1
			}
			else_if = { 
				limit = { mil = 8 }
				change_dip = 1
			}
			else = {
				random_list = {
					1 = { change_dip = 1 }
					1 = { change_mil = 1 }
				}
			}
		}
		else_if = {
			limit = { dip = 8 }
			if = {
				limit = { mil = 8}
				change_adm = 1
			}
			else = {
				random_list = {
					1 = { change_adm = 1 }
					1 = { change_mil = 1 }
				}
			}
		}
		else_if = {
			limit = { mil = 8 }
			random_list = {
				1 = { change_adm = 1 }
				1 = { change_dip = 1 }
			}
		}
		else = {
			random_list = {
				1 = { change_adm = 1 }
				1 = { change_dip = 1 }
				1 = { change_mil = 1 }
			}
		}
	}
}
